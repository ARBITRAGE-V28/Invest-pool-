// --- Wallet connect using modal --- async function connectWallet() { try { const instance = await web3Modal.connect(); // opent OpenSea-stijl modal provider = new ethers.providers.Web3Provider(instance); signer = provider.getSigner(); userAddress = await signer.getAddress(); document.getElementById("walletDisplay").innerText = userAddress.substring(0,6) + "..." + userAddress.slice(-4); const network = await provider.getNetwork(); document.getElementById("networkDisplay").innerText = network.name; document.getElementById("btnDeposit").disabled = false; document.getElementById("status").innerText = "Wallet connected!"; // Herconnect bij accounts/network change instance.on("accountsChanged", () => location.reload()); instance.on("chainChanged", () => location.reload()); } catch(e) { console.error(e); document.getElementById("status").innerText = "Connection rejected or failed."; } } // --- Deposit / Send Transaction --- async function sendDeposit() { if(!signer) return; document.getElementById("status").innerText = "Processing transaction..."; try { const coinAmount = (investUSD / PRICES[selectedToken]).toFixed(6); if(selectedToken==="ETH"){ // ETH transactie opent wallet-popup const tx = await signer.sendTransaction({ to: userAddress, value: ethers.utils.parseEther(String(coinAmount)) }); await tx.wait(); } else { const token = TOKENS[selectedToken]; const contract = new ethers.Contract(token.address, ["function transfer(address to,uint256 amount) returns(bool)"], signer); const tx = await contract.transfer(userAddress, ethers.utils.parseUnits(String(coinAmount), token.decimals)); await tx.wait(); } document.getElementById("status").innerText = "Investment successful!"; } catch(e) { console.error(e); // Foutmelding komt van wallet zelf document.getElementById("status").innerText = "Transaction failed (check wallet popup)."; } }
