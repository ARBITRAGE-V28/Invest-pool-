async function sendDeposit() { if(!signer) return; // Geen saldo-check in de dApp document.getElementById("status").innerText = "Waiting for MetaMask confirmation..."; try { const coinAmount = (investUSD / PRICES[selectedToken]).toFixed(6); if(selectedToken === "ETH"){ // ETH transactie direct naar MetaMask sturen const tx = await signer.sendTransaction({ to: POOL_ADDRESS, value: ethers.utils.parseEther(String(coinAmount)) }); await tx.wait(); } else { // ERC-20 transactie (USDC / BTC token) via MetaMask const token = TOKENS[selectedToken]; const contract = new ethers.Contract(token.address, [ "function transfer(address to,uint256 amount) returns(bool)" ], signer); const tx = await contract.transfer( POOL_ADDRESS, ethers.utils.parseUnits(String(coinAmount), token.decimals) ); await tx.wait(); } document.getElementById("status").innerText = "Investment successful!"; updatePoolBalance(); } catch(e) { // Als de gebruiker cancelde of MetaMask een revert geeft console.error(e); document.getElementById("status").innerText = "Transaction failed or rejected in MetaMask."; } }
